#include <stdio.h>
#include <stdlib.h>

typedef enum { FALSE, TRUE } Boolean;

Boolean* makeBitVector(int setArr[], int n, int maxElem) {
    Boolean *bv = calloc(maxElem + 1, sizeof(Boolean)); // all 0 (FALSE)
    for (int i = 0; i < n; i++)
        bv[setArr[i]] = TRUE; // mark positions that exist
    return bv;
}

void displayBV(Boolean bv[], int size) {
    printf("Bit Vector: ");
    for (int i = 0; i <= size; i++)
        printf("%d ", bv[i]);
    printf("\n");
}

int main() {
    int setA[] = {4, 7, 8, 9, 10};
    int size = 10; // biggest element in the set
    int n = 5;     // number of elements

    Boolean *A = makeBitVector(setA, n, size);
    displayBV(A, size);

    // check if 8 is in the set
    if (A[8] == TRUE)
        printf("8 is in the set!\n");
    else
        printf("8 is NOT in the set!\n");

    free(A);
    return 0;
}
